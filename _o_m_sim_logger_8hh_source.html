61  \
62  /* display the function doing the logging */ \

64  \
65  /* display the file and/or the line number */ \
66  printf( \

71  DISPLAY_LINE ? line : 0, \

73  !DISPLAY_FUNC && (DISPLAY_FILE || DISPLAY_LINE) ? " " : "" \
74  ); \
75  \

77  printf("%s%s", DISPLAY_BORDER ? BORDER : "", DISPLAY_BORDER ? " " : ""); \
78  \

80  if (DISPLAY_MESSAGE) printf(__VA_ARGS__); \
